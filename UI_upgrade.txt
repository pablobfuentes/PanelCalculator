// ── STATE ────────────────────────────────────────────────
const state = {
  canvasW: 15, canvasH: 12,
  gap: 1, reach: 3, alleyW: 1,
  spineEdge: 'bottom',
  locked: false,
  selectedPanel: 'A',
  panels: {
    A: { name:'Panel A', length:2.3, width:1.2, weight:25, wp:540, color:'#3B7DD8', emoji:'🔲' },
    B: { name:'Panel B', length:2.0, width:1.0, weight:20, wp:450, color:'#3ECF8E', emoji:'🟩' },
  }
};

const ALLEY_COLOR    = 'rgba(245,163,35,0.65)';
const ALLEY_STROKE   = '#F5A623';
const PANEL_COLORS   = { A:'#3B7DD8', B:'#3ECF8E' };
const PANEL_STROKES  = { A:'#5BB8F5', B:'#62E8AC' };
const SPINE_COLOR    = 'rgba(245,163,35,0.4)';
const SPINE_STROKE   = '#F5A623';
const GAP_IN_TO_M    = 0.0254;

// ── CANVAS SETUP ─────────────────────────────────────────
const canvas = document.getElementById('solarCanvas');
const ctx    = canvas.getContext('2d');

// ── LAYOUT CALC ──────────────────────────────────────────
function computeLayout() {
  const p = state.panels[state.selectedPanel];
  const pW = p.width;   // panel width (columns direction)
  const pH = p.length;  // panel height (rows direction)
  const gapM = state.gap * GAP_IN_TO_M;
  const aW   = state.alleyW;
  const reach = state.reach;
  const cW    = state.canvasW;
  const cH    = state.canvasH;

  // Pair width = 2 * pW + gap
  const pairW = 2 * pW + gapM;

  // How many pairs fit per group (group = 2*reach panels = reach pairs)
  // Alley reach is in panels, group = 2*reach panels wide
  // We use pairs as units: groupPairs = reach (since 1 pair = 2 panels)
  const groupPairs = Math.ceil(reach / 2);

  // Alley columns: after every groupPairs pairs, insert an alley
  // First: how many pairs fit total?
  // Total width = numPairs * pairW + numAlleys * aW + spineAW (spine is along Y, so adds to height)
  // Iteratively find numPairs
  let numPairs = Math.floor(cW / pairW);
  while (numPairs > 0) {
    const numAlleys = Math.floor((numPairs - 1) / groupPairs);
    const usedW = numPairs * pairW + numAlleys * aW;
    if (usedW <= cW) break;
    numPairs--;
  }
  const numPairsVal = Math.max(1, numPairs);
  const numAlleysX  = Math.floor((numPairsVal - 1) / groupPairs);

  // Rows: spine alley is horizontal (along X), adds to height
  const spineH = aW;
  let numRows = Math.floor((cH - spineH) / pH);
  numRows = Math.max(1, numRows);

  const totalPanels = numPairsVal * 2 * numRows;
  const usedW = numPairsVal * pairW + numAlleysX * aW;
  const usedH = numRows * pH + spineH;

  return { pW, pH, gapM, pairW, aW, reach,
    numPairs: numPairsVal, numAlleys: numAlleysX,
    numRows, totalPanels, usedW, usedH, spineH, groupPairs };
}

// ── DRAW ─────────────────────────────────────────────────
let animFrame = null;
let _targetLayout = null;
let _currentScale = 1;

function drawLayout(animPct) {
  const l = computeLayout();
  _targetLayout = l;

  const p = state.panels[state.selectedPanel];
  const MARGIN = 10;
  const maxW   = canvas.width  - MARGIN * 2;
  const maxH   = canvas.height - MARGIN * 2;
  const scaleX = maxW / l.usedW;
  const scaleY = maxH / l.usedH;
  const scale  = Math.min(scaleX, scaleY);
  _currentScale = scale;

  const drawW = l.usedW * scale;
  const drawH = l.usedH * scale;
  const ox = (canvas.width  - drawW) / 2;
  const oy = (canvas.height - drawH) / 2;

  const pColor = PANEL_COLORS[state.selectedPanel] || p.color || '#3B7DD8';
  const sColor = PANEL_STROKES[state.selectedPanel] || '#5BB8F5';

  ctx.clearRect(0, 0, canvas.width, canvas.height);

  // Canvas background
  ctx.fillStyle = '#111827';
  ctx.beginPath();
  ctx.roundRect(ox - 4, oy - 4, drawW + 8, drawH + 8, 6);
  ctx.fill();

  // Grid ghost lines
  ctx.strokeStyle = 'rgba(255,255,255,0.03)';
  ctx.lineWidth = 1;
  for (let gx = ox; gx < ox + drawW; gx += 20) {
    ctx.beginPath(); ctx.moveTo(gx, oy); ctx.lineTo(gx, oy + drawH); ctx.stroke();
  }
  for (let gy = oy; gy < oy + drawH; gy += 20) {
    ctx.beginPath(); ctx.moveTo(ox, gy); ctx.lineTo(ox + drawW, gy); ctx.stroke();
  }

  // Spine alley (horizontal, full width)
  const spineY = state.spineEdge === 'top'
    ? oy
    : oy + drawH - l.spineH * scale;
  const panelAreaY = state.spineEdge === 'top'
    ? oy + l.spineH * scale
    : oy;

  ctx.fillStyle = ALLEY_COLOR;
  ctx.strokeStyle = ALLEY_STROKE;
  ctx.lineWidth = 1;
  ctx.beginPath();
  ctx.roundRect(ox, spineY, drawW, l.spineH * scale, 3);
  ctx.fill(); ctx.stroke();

  // Spine label
  ctx.save();
  ctx.font = `bold ${Math.max(8, 9 * (scale/25))}px Rajdhani`;
  ctx.fillStyle = 'rgba(245,163,35,0.9)';
  ctx.textAlign = 'center';
  ctx.textBaseline = 'middle';
  ctx.fillText('SPINE ALLEY — EDGE ACCESS', ox + drawW / 2, spineY + (l.spineH * scale) / 2);
  ctx.restore();

  // Draw pairs and vertical alleys
  const groupPairs = l.groupPairs;
  let cx = ox;
  for (let pi = 0; pi < l.numPairs; pi++) {
    // Insert vertical alley before this pair (except first)?
    if (pi > 0 && pi % groupPairs === 0) {
      // Draw alley
      ctx.fillStyle = ALLEY_COLOR;
      ctx.strokeStyle = ALLEY_STROKE;
      ctx.lineWidth = 1;
      ctx.beginPath();
      ctx.roundRect(cx, panelAreaY, l.aW * scale, l.numRows * l.pH * scale, 3);
      ctx.fill(); ctx.stroke();

      // Alley label (rotated)
      const alleyIdx = Math.floor(pi / groupPairs);
      ctx.save();
      ctx.translate(cx + (l.aW * scale) / 2, panelAreaY + (l.numRows * l.pH * scale) / 2);
      ctx.rotate(-Math.PI / 2);
      ctx.font = `700 ${Math.max(7, 8 * (scale/25))}px Rajdhani`;
      ctx.fillStyle = 'rgba(245,163,35,0.85)';
      ctx.textAlign = 'center';
      ctx.textBaseline = 'middle';
      ctx.fillText('A' + alleyIdx, 0, 0);
      ctx.restore();

      cx += l.aW * scale;
    }

    // Draw rows for this pair
    for (let ri = 0; ri < l.numRows; ri++) {
      const py = panelAreaY + ri * l.pH * scale;
      const pw = l.pW * scale;
      const ph = l.pH * scale;

      // Left panel
      ctx.fillStyle = pColor;
      ctx.strokeStyle = sColor;
      ctx.lineWidth = 0.5;
      ctx.beginPath();
      ctx.roundRect(cx, py, pw, ph, 2);
      ctx.fill(); ctx.stroke();
      // Cell line
      ctx.strokeStyle = 'rgba(255,255,255,0.12)';
      ctx.lineWidth = 0.5;
      ctx.beginPath();
      ctx.moveTo(cx + 1, py + ph/2); ctx.lineTo(cx + pw - 1, py + ph/2);
      ctx.stroke();

      // Gap
      const gw = l.gapM * scale;
      if (gw > 0.5) {
        ctx.fillStyle = 'rgba(255,193,7,0.3)';
        ctx.fillRect(cx + pw, py, Math.max(gw, 1), ph);
      }

      // Right panel
      ctx.fillStyle = pColor;
      ctx.strokeStyle = sColor;
      ctx.lineWidth = 0.5;
      ctx.beginPath();
      ctx.roundRect(cx + pw + gw, py, pw, ph, 2);
      ctx.fill(); ctx.stroke();
      ctx.strokeStyle = 'rgba(255,255,255,0.12)';
      ctx.lineWidth = 0.5;
      ctx.beginPath();
      ctx.moveTo(cx + pw + gw + 1, py + ph/2); ctx.lineTo(cx + pw + gw + pw - 1, py + ph/2);
      ctx.stroke();
    }
    cx += l.pairW * scale;
  }

  // Border glow
  ctx.strokeStyle = 'rgba(91,184,245,0.25)';
  ctx.lineWidth = 1.5;
  ctx.strokeRect(ox - 4, oy - 4, drawW + 8, drawH + 8);

  // Update UI stats
  document.getElementById('panel-count').textContent = l.totalPanels;
  document.getElementById('area-val').textContent =
    l.usedW.toFixed(1) + ' × ' + l.usedH.toFixed(1) + ' m';
  document.getElementById('area-sub').textContent =
    (l.usedW * l.usedH).toFixed(1) + ' m²';
}

// ── CONTROLS ─────────────────────────────────────────────
const limits = { gap:[1,4], reach:[2,4], alleyW:[1,3] };

function adj(key, delta) {
  const [lo, hi] = limits[key];
  state[key] = Math.max(lo, Math.min(hi, state[key] + delta));
  document.getElementById('val-' + key).textContent = state[key];
  drawAllTips();
  schedDraw();
}

function toggleSpine() {
  state.spineEdge = state.spineEdge === 'bottom' ? 'top' : 'bottom';
  document.getElementById('spine-btn').textContent = state.spineEdge.toUpperCase();
  drawAllTips();
  schedDraw();
}

function toggleLock() {
  state.locked = !state.locked;
  const btn = document.getElementById('lock-btn');
  btn.textContent = state.locked ? '🔒' : '🔓';
  btn.classList.toggle('locked', state.locked);
  showToast(state.locked ? 'Panel count locked' : 'Panel count unlocked');
}

function selectPanel(key) {
  state.selectedPanel = key;
  document.querySelectorAll('.panel-type-btn').forEach(b => b.classList.remove('active'));
  document.getElementById('btn-' + key)?.classList.add('active');
  const p = state.panels[key];
  if (p) {
    document.getElementById('panel-name').textContent = p.name;
    document.getElementById('p-length').textContent = p.length + ' m';
    document.getElementById('p-width').textContent  = p.width  + ' m';
    document.getElementById('p-weight').textContent = p.weight + ' kg';
    document.getElementById('p-wp').textContent     = p.wp     + ' Wp';
    document.getElementById('panel-icon').textContent = p.emoji || '🔲';
    PANEL_COLORS[key] = p.color;
    PANEL_STROKES[key] = lighten(p.color);
  }
  schedDraw();
}

function lighten(hex) {
  const r = parseInt(hex.slice(1,3),16), g = parseInt(hex.slice(3,5),16), b = parseInt(hex.slice(5,7),16);
  return `rgb(${Math.min(255,r+60)},${Math.min(255,g+60)},${Math.min(255,b+60)})`;
}

function addPanel() {
  const name = document.getElementById('modal-name').value || 'Panel C';
  const len  = parseFloat(document.getElementById('modal-len').value)  || 2.0;
  const wid  = parseFloat(document.getElementById('modal-wid').value)  || 1.0;
  const kg   = parseFloat(document.getElementById('modal-kg').value)   || 20;
  const wp   = parseFloat(document.getElementById('modal-wp').value)   || 400;

  // Find next available key
  const used = Object.keys(state.panels);
  const key  = PANEL_KEYS.find(k => !used.includes(k)) || 'Z';

  state.panels[key] = { name, length: len, width: wid, weight: kg, wp, color: selectedColor, emoji: '🟥' };
  PANEL_COLORS[key]  = selectedColor;
  PANEL_STROKES[key] = lighten(selectedColor);

  // Add button
  const row = document.querySelector('.panel-type-row');
  const addBtn = document.querySelector('.add-btn');
  const btn = document.createElement('div');
  btn.className = 'panel-type-btn';
  btn.id = 'btn-' + key;
  btn.title = name;
  btn.textContent = key;
  btn.style.background = selectedColor + '33';
  btn.style.borderColor = selectedColor + '99';
  btn.style.color = selectedColor;
  btn.onclick = () => selectPanel(key);
  row.insertBefore(btn, addBtn);

  closeModal();
  selectPanel(key);
  showToast(name + ' added!');
}
